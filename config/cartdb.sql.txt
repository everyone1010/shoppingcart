CREATE TABLE `User` (
`FIRST_NAME` VARCHAR(100) NOT NULL,
`LAST_NAME` VARCHAR(100) NOT NULL,
`MIDDLE_NAME` VARCHAR(100),
`DOB` DATE NOT NULL,
`EMAIL` VARCHAR(100) NOT NULL,
`CONTACT_NO` VARCHAR(100) NOT NULL,
`PASSWORD` VARCHAR(50) NOT NULL,
`SECURITY_QUESTION` VARCHAR(100) NOT NULL,
`SECURITY_QUESTION_ANSWER` VARCHAR(100) NOT NULL,
`CITY` VARCHAR(100) NOT NULL,
`STREET_ONE` VARCHAR(100) NOT NULL,
`STREET_TWO` VARCHAR(100),
`COUNTRY` VARCHAR(100) NOT NULL,
`PIN` VARCHAR(20) NOT NULL,
`STATE` VARCHAR(50) NOT NULL,
`USER_ID` VARCHAR(50) NOT NULL,
`LAST_LOGGED_IN` DATETIME,
PRIMARY KEY (`USER_ID`)
);

CREATE TABLE `Seller` (
`FIRST_NAME` VARCHAR(100) NOT NULL,
`LAST_NAME` VARCHAR(100) NOT NULL,
`MIDDLE_NAME` VARCHAR(100) NOT NULL,
`CONTACT_NO` VARCHAR(100) NOT NULL,
`CITY` VARCHAR(100) NOT NULL,
`STREET_ONE` VARCHAR(100) NOT NULL,
`STREET_TWO` VARCHAR(100),
`COUNTRY` VARCHAR(100) NOT NULL,
`PIN` VARCHAR(20) NOT NULL,
`STATE` VARCHAR(50) NOT NULL,
`SELLER_ID` VARCHAR(50) NOT NULL,
PRIMARY KEY (`SELLER_ID`)
);

CREATE TABLE `Product` (
`PRODUCT_NAME` VARCHAR(50) NOT NULL,
`PRODUCT_ID` INT NOT NULL AUTO_INCREMENT,
`MANUFACTURER` VARCHAR(50) NOT NULL,
`CATEGORY` VARCHAR(50) NOT NULL,
`SELLER_PRODUCT_ID` VARCHAR(100) NOT NULL,
`PRICE` FLOAT(10) NOT NULL,
`SELLER_USER_ID` VARCHAR(50) NOT NULL,
`SELLER_ID` VARCHAR(50) NOT NULL,
`VENDOR` VARCHAR(50) NOT NULL,
`ADDITION_TIME` DATETIME NOT NULL,
`CURRENCY` varchar(10) NOT NULL,
`ALREADY_SOLD` BINARY NOT NULL DEFAULT false,
PRIMARY KEY (`PRODUCT_ID`)
);

CREATE TABLE `Cart` (
`USER_ID` VARCHAR(50),
`PRODUCT_ID` INT NOT NULL,
`TOKEN` VARCHAR(4000),
`CART_ID` INT NOT NULL AUTO_INCREMENT,
`TYPE` varchar(20) NOT NULL,
PRIMARY KEY (`CART_ID`)
);

CREATE TABLE `Payment` (
`CARD_NUMBER` VARCHAR(50),
`PAYMENT_DATE` DATETIME NOT NULL,
`AMOUNT` FLOAT(10) NOT NULL,
`PAYMENT_ID` INT NOT NULL AUTO_INCREMENT,
`ACCOUNT_HOLDER` VARCHAR(100) NOT NULL,
`CATEGORY` VARCHAR(100) NOT NULL,
`BANK_NAME` VARCHAR(100),
`BANK_IFSC` VARCHAR(100),
`CURRENCY` varchar(100) NOT NULL,
PRIMARY KEY (`PAYMENT_ID`)
);

CREATE TABLE `Invoice` (
`PAYMENT_ID` INT NOT NULL,
`PRODUCT_ID` INT NOT NULL,
`CONFIRMATION_NUMBER` INT NOT NULL AUTO_INCREMENT,
PRIMARY KEY (`CONFIRMATION_NUMBER`)
);

ALTER TABLE `Product` ADD CONSTRAINT `Product_fk0` FOREIGN KEY (`SELLER_ID`) REFERENCES `Seller`(`SELLER_ID`);

ALTER TABLE `Cart` ADD CONSTRAINT `Cart_fk0` FOREIGN KEY (`PRODUCT_ID`) REFERENCES `Product`(`PRODUCT_ID`);

ALTER TABLE `Invoice` ADD CONSTRAINT `Invoice_fk0` FOREIGN KEY (`PAYMENT_ID`) REFERENCES `Payment`(`PAYMENT_ID`);

ALTER TABLE `Invoice` ADD CONSTRAINT `Invoice_fk1` FOREIGN KEY (`PRODUCT_ID`) REFERENCES `Product`(`PRODUCT_ID`);
